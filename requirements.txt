# Requirements for NewCaptionApp - Video Captioner
# Install with: pip install -r requirements.txt

# IMPORTANT: Requires Python 3.11 or 3.12
# - Python 3.13+ is too new (PyTorch 2.8.0 wheels not available)
# - Python 3.10 or older is too old
# Recommended: Python 3.12.x

# GUI Framework
PyQt5>=5.15.0,<6.0.0

# Deep Learning Framework
# Pinned to 2.8.0 to avoid WinError 1114 bug in PyTorch 2.9.0+ on Windows
# See: https://github.com/pytorch/pytorch/issues/166628
# PyTorch 2.8.0 is compatible with Python 3.8-3.12 only
torch==2.8.0
torchvision==0.23.0
torchaudio==2.8.0

# Speech Recognition
faster-whisper>=1.0.0

# Audio Processing
# Librosa removed - using FFmpeg directly for better performance

# Additional dependencies that will be installed automatically:
# - numpy (required by torch)
# - ctranslate2 (required by faster-whisper)
# - ctranslate2 (required by faster-whisper)
# - tokenizers (required by faster-whisper)
# - onnxruntime (may be required by faster-whisper)
# - huggingface-hub (required by faster-whisper)

# Note: FFmpeg is required but must be installed separately as a system dependency
# Install FFmpeg:
# - macOS: brew install ffmpeg
# - Ubuntu/Debian: sudo apt-get install ffmpeg
# - Windows: Download from https://ffmpeg.org/download.html

# ──────────────────────────────────────────────────────────────────────────────
# TRAINING DEPENDENCIES (fine-tuning only — not required at runtime)
# Install with: pip install -r requirements.txt --extra-index-url ...
# Or install the training group separately:
#   pip install transformers datasets evaluate jiwer accelerate peft soundfile tensorboard ctranslate2
# ──────────────────────────────────────────────────────────────────────────────

# HuggingFace Transformers — WhisperForConditionalGeneration, Seq2SeqTrainer

# TensorBoard — training loss and WER visualisation

tensorboard>=2.14.0

# CTranslate2 — post-training model conversion (HuggingFace → faster-whisper format)
# Provides the `ct2-transformers-converter` CLI command used by convert_model.py
ctranslate2>=4.0.0
